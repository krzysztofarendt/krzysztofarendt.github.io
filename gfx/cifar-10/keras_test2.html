def unpickle(file):
    import pickle
    with open(file, 'rb') as fo:
        dict = pickle.load(fo, encoding='bytes')
    return dict


def get_batch(batch):
    """
    :param batch: str, 'data_batch_1', ..., 'data_batch_5', 'test_batch'
    :return: dict
    """
    dct_copy = unpickle("cifar-10-batches-py/{}".format(batch))

    dct = dict()
    for key in dct_copy:
        dct[key.decode('utf-8')] = dct_copy[key]
    
    return dct


def get_all():
    """
    The function does the following:
    
    1. Reads all five batches and puts all images in an array
       x_train of shape (50000, 32, 32, 3).
    2. Reads the respective labels and puts them into an array
       y_train of shape (50000, 1).
    3. Reads the test batch and puts all images in an array
       x_test of shape (10000, 32, 32, 3).
    4. Reads the respective labels and puts them into an array
       y_test of shape (10000, 1)
        
    :return: x_train, y_train, x_text, y_test
    """
    
    train_batches = ['data_batch_{}'.format(x) for x in range(1, 6)]
    test_batch = 'test_batch'
    
    train_dct = [get_batch(x) for x in train_batches]
    test_dct = get_batch(test_batch)
    
    def flat_2_im(flat):
        im_r = flat[:1024].reshape((32, 32))
        im_g = flat[1024:1024+32*32].reshape((32, 32))
        im_b = flat[2048:2048+32*32].reshape((32, 32))

        im = np.zeros(shape=(32, 32, 3))
        im[:,:,0] = im_r
        im[:,:,1] = im_g
        im[:,:,2] = im_b

        im /= 255.
        
        return im
    
    # x_train
    x_train = np.zeros((50000, 32, 32, 3))
    n = 0
    for d in train_dct:
        # Get image i
        for i in range(d['data'].shape[0]):
            # Flat image to 32x32x3, normalized to 0...1
            flat = d['data'][i]
            im = flat_2_im(flat)
            
            # Add to x_train
            x_train[n, :, :, :] = im
            n += 1
    
    # y_train
    y_train = np.zeros((50000, 1))
    
    n = 0
    for d in train_dct:
        y_train[n:n+len(d['labels']), 0] = d['labels']
        n += len(d['labels'])
    
    # x_test
    x_test = np.zeros((10000, 32, 32, 3))
    
    for i in range(test_dct['data'].shape[0]):
        flat = test_dct['data'][i]
        im = flat_2_im(flat)
        x_test[i, :, :, :] = im
        
    # y_test
    y_test = np.array(test_dct['labels']).reshape((10000, 1))
    
    return x_train, y_train, x_test, y_test

def visualize_filter(filter_index, return_history=False):
    """
    :param filter_index: int, 0..31
    :return: image as array (32, 32, 3)
    """
    layer_name = 'conv2d_1'
    img_width, img_height = 32, 32
    step = 0.05

    # Placeholder for input tensor
    input_img = model2.input

    # Build a loss function that maximizes the activation
    # of the nth filter of the layer considered
    layer_output = layer_dict[layer_name].output
    loss = K.mean(layer_output[:, :, :, filter_index])

    # Compute the gradient of the input picture wrt this loss
    grads = K.gradients(loss, input_img)[0]

    # Normalization trick: we normalize the gradient
    grads /= (K.sqrt(K.mean(K.square(grads))) + K.epsilon())

    # This function returns the loss and grads given the input picture
    iterate = K.function(inputs=[input_img], outputs=[loss, grads])

    # We start from a gray image with some noise
    input_img_data = np.random.random((1, img_width, img_height, 3))

    # Run gradient ascent for 50 steps 
    # (number of steps has a limited influence,
    #  as the image is later normalized anyway)
    loss_history = list()
    
    for i in range(20):
        loss_value, grads_value = iterate([input_img_data])
        input_img_data += grads_value * step
        loss_history.append(loss_value)

    # Util function to convert a tensor into a valid image
    def deprocess_image(x):
        # normalize tensor: center on 0., ensure std is 0.1
        x -= x.mean()
        x /= (x.std() + 1e-5)
        x *= 0.2

        # recenter on 0.5, clip to [0, 1]
        x += 0.5
        x = np.clip(x, 0, 1)
        return x

    img = input_img_data[0]
    img = deprocess_image(img)
    
    if return_history:
        return img, loss_history
    else:
        return img

def find_ideal_image(label, return_history=False):
    """
    :param label: int, 0..9
    :return: image as array (32, 32, 3)
    """
    img_width, img_height = 32, 32
    step = 5.

    # Placeholder for input tensor
    input_img = model2.input

    # Build a loss function that maximizes
    # the activation of the nth label
    loss = K.mean(model2.output[:, label])

    # Compute the gradient of the input picture wrt this loss
    grads = K.gradients(loss, input_img)[0]

    # Normalization trick: we normalize the gradient
    grads /= (K.sqrt(K.mean(K.square(grads))) + K.epsilon())

    # This function returns the loss and grads given the input picture
    iterate = K.function(inputs=[input_img], outputs=[loss, grads])

    # We start from a gray image with some noise
    input_img_data = np.random.random((1, img_width, img_height, 3))

    # Run gradient ascent for 50 steps 
    # (number of steps doesn't really matter
    # as the image is later normalized anyway)
    loss_history = list()
    
    for i in range(20):
        loss_value, grads_value = iterate([input_img_data])
        input_img_data += grads_value * step
        loss_history.append(loss_value)

    # Util function to convert a tensor into a valid image
    def deprocess_image(x):
        # normalize tensor: center on 0., ensure std is 0.1
        x -= x.mean()
        x /= (x.std() + 1e-5)
        x *= 0.15

        # recenter on 0.5, clip to [0, 1]
        x += 0.5
        x = np.clip(x, 0, 1)
        return x

    img = input_img_data[0]
    img = deprocess_image(img)
    
    if return_history:
        return img, loss_history
    else:
        return img
